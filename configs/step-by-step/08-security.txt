======= 1. Extended ACLs =======
## for DSW3 & DSW4 ##
ip access-list extended PROTECT_SERVERS
permit ip 10.1.100.0 0.0.0.255 10.1.120.0 0.0.0.255
permit udp 10.0.0.0 0.255.255.255 10.1.120.20 0.0.0.0 eq 67
permit udp 10.0.0.0 0.255.255.255 10.1.120.20 0.0.0.0 eq 68
permit udp 10.0.0.0 0.255.255.255 10.1.120.30 0.0.0.0 eq 53
permit tcp 10.0.0.0 0.255.255.255 10.1.120.30 0.0.0.0 eq 53
permit tcp 10.0.0.0 0.255.255.255 10.1.120.40 0.0.0.0 eq 80
permit tcp 10.0.0.0 0.255.255.255 10.1.120.40 0.0.0.0 eq 443
permit udp 10.0.0.0 0.255.255.255 10.1.120.40 0.0.0.0 eq 514
deny ip 10.0.0.0 0.255.255.255 10.1.120.0 0.0.0.255
permit ip any any
exit
interface vlan 120
ip access-group PROTECT_SERVERS out

======= 2. Port Security =======
## for ASW1-AW11 ##
interface f0/1
switchport port-security
switchport port-security maximum 3
switchport port-security mac-address sticky
switchport port-security violation restrict
interface f0/23
switchport port-security
switchport port-security maximum 1
switchport port-security mac-address sticky
switchport port-security violation restrict

## for ASW12 ##
interface range f0/1-5
switchport port-security
switchport port-security maximum 1
switchport port-security mac-address sticky
switchport port-security violation restrict
exit
do copy run start

======= 3. DHCP Snooping =======
## for ASW1-ASW6 ##
ip dhcp snooping
ip dhcp snooping vlan 10,20,30,40,50,60,200
no ip dhcp snooping information option
interface range g0/1-2
ip dhcp snooping trust
interface f0/24
ip dhcp snooping limit rate 100
interface range f0/1,f0/23
ip dhcp snooping limit rate 15

## for ASW7-ASW11 ##
