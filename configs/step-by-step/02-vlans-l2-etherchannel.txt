======= 1. L2 Etherchannel =======

======= 2. Trunk Configuration =======
## for ASW1-ASW6 ##
interface range g0/1-2
switchport mode trunk
switchport nonegotiate
switchport trunk native vlan 1000
switchport trunk allowed vlan 10,20,30,40,50,60,200,500

## for ASW7-ASW12 ##
interface range g0/1-2
switchport mode trunk
switchport nonegotiate
switchport trunk native vlan 1000
switchport trunk allowed vlan 70,80,90,100,110,120,200,500

## for DSW1 & DSW2 ##
interface range g1/0/1-6
switchport mode trunk
switchport nonegotiate
switchport trunk native vlan 1000
switchport trunk allowed vlan 10,20,30,40,50,60,200,500
exit
interface po1
switchport mode trunk
switchport nonegotiate
