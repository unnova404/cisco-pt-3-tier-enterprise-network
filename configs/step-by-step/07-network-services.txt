======= 8. SSH =======
## for all network devices ##
crypto key generate rsa
4096
ip ssh version 2
ip access-list standard VTY_SSH
permit 10.1.100.0 0.0.0.255
exit
line vty 0 15
access-class VTY_SSH in
transport input ssh
login local
logging synchronous

======= 9. Dynamic PAT =======
## for R1 ##
ip access-list standard PAT-ACL
permit 10.0.0.0 0.0.0.255
permit 10.1.0.0 0.0.0.255
permit 10.1.10.0 0.0.0.255
permit 10.1.20.0 0.0.0.255
permit 10.1.30.0 0.0.0.255
permit 10.1.40.0 0.0.0.255
permit 10.1.50.0 0.0.0.255
permit 10.1.60.0 0.0.0.255
permit 10.1.70.0 0.0.0.255
permit 10.1.80.0 0.0.0.255
permit 10.1.90.0 0.0.0.255
permit 10.1.100.0 0.0.0.255
permit 10.1.110.0 0.0.0.255
permit 10.1.120.0 0.0.0.255
permit 10.1.200.0 0.0.0.255
permit 10.1.500.0 0.0.0.255
exit
ip nat inside source list PAT-ACL interface g0/0/0 overload
interface range g0/1/0,g0/2/0
ip nat inside
interface g0/0/0
ip nat outside
exit
do copy run start
